Context
Playing full games to evaluate agent strength takes ~50 seconds per run (testslow.js).
A puzzle benchmark presents positions with one correct move and checks whether each
agent finds it. Much faster, and provides a direct signal about tactical reading.
The system should support puzzles at any board size, and use a data-driven format so
positions can be added easily — whether hand-crafted or auto-generated in the future
(e.g., by having the strongest agent play itself and harvesting positions where one
move is overwhelmingly preferred).
Design
Data format — board as ASCII string
Each puzzle is specified with a human-readable board diagram:
jsconst PUZZLES = [
  {
    name: 'Capture last liberty',
    toPlay: 'black',
    answer: [[2,1]],          // acceptable [x,y] answers
    board: `
      · · · · ·

  },
  {

    toPlay: 'black',

      · · · · · · ·
      · · · ○ · · ·

      · · · · · · ·
      · · · · · · ·
      · · · · · · ·
    `,

● = black, ○ = white, · = empty (same chars as selfplay.js output)
Board size is inferred from the diagram dimensions (rows × cols must be square)
answer is a list of [x, y] pairs (usually one; multiple for puzzles
where several moves are equally correct)
Different puzzles can use different board sizes

Parser
jsfunction parseBoard(boardStr) {
  const rows = boardStr.trim().split('\n').map(r => r.trim().split(/\s+/));
  const size = rows.length;
  // rows.length must equal rows[0].length (square board)
  const stones = [];
  for (let y = 0; y < size; y++)
    for (let x = 0; x < size; x++) {
      const c = rows[y][x];
      if (c === '●') stones.push([x, y, 'black']);
      else if (c === '○') stones.push([x, y, 'white']);
    }
  return { size, stones };
}
Position setup helper
jsfunction buildPosition(puzzle) {
  const { size, stones } = parseBoard(puzzle.board);
  const game = new Game(size, 0);
  const c = size >> 1;
  game.board.set(c, c, null);   // clear auto-placed center stone
  game.hash = 0n;
  game.moveCount = 0;
  game.current = puzzle.toPlay;
  game.consecutivePasses = 0;
  game.koFlag = null;
  for (const [x, y, color] of stones) {
    game.board.set(x, y, color);
    game.hash ^= ZOBRIST[y][x][color];
  }
  game.board._rebuildGroups();
  return game;
}
```

Key files: `game.js` (`Game`, `Board`, `ZOBRIST`, `_rebuildGroups`), `ai/*.js`.

### Puzzle set

Start with ~12 puzzles across 5x5 and 7x7 boards:

**5x5 puzzles (6-7):**
1. Capture single stone (last liberty)
2. Capture two-stone group
3. Save own group from atari (extend)
4. Capture to save (remove attacker)
5. Don't self-atari (one safe move among several)
6. Connect two groups
7. Toroidal wrap capture (group at edge, liberty wraps to opposite side)

**7x7 puzzles (5-6):**
8. Capture a 3-stone group
9. Save a larger group from atari
10. Double atari (one move threatens two groups)
11. Capture vs extend (capturing is clearly better than defending)
12. Toroidal wrap — group spanning the board edge

Each puzzle verified during implementation: build position, confirm expected move
is legal and clearly best on the toroidal board.

### CLI interface
```
node testpuzzles.js --agent <name> [--budget <ms>]

--agent (required): agent name (random, mc, mcts, amaf)
--budget (optional, default 200): time budget in ms per puzzle

Parse args with the same minimal approach as selfplay.js (manual argv parsing).

Round-trip test in testfast.js
Add a test to testfast.js that verifies parseBoard and boardToString are
consistent: create a Game, play some random moves, serialize the board to the
● ○ · format, parse it back, and confirm every cell matches.
This requires parseBoard and boardToString to be importable. Two options:

Put them in testpuzzles.js and also export them (UMD guard)
Put them in a shared utility file

Since testpuzzles.js already needs them and they're small, put them in
testpuzzles.js with a module.exports at the bottom. Then testfast.js
does const { parseBoard, boardToString } = require('./testpuzzles.js').
js// In testpuzzles.js:
function boardToString(board) {
  const rows = [];
  for (let y = 0; y < board.size; y++) {
    const cells = [];
    for (let x = 0; x < board.size; x++) {
      const v = board.get(x, y);
      cells.push(v === 'black' ? '●' : v === 'white' ? '○' : '·');
    }
    rows.push(cells.join(' '));
  }
  return rows.join('\n');
}

// At bottom:
if (typeof module !== 'undefined') module.exports = { parseBoard, boardToString };

section('Board serialize/parse round-trip');
{
  const { parseBoard, boardToString } = require('./testpuzzles.js');
  const g = new Game(7, 3.5);
  // play some random moves
  for (let i = 0; i < 10 && !g.gameOver; i++) { ... }
  const str = boardToString(g.board);
  const { size, stones } = parseBoard(str);
  assert(size === 7, 'parsed size matches');
  // rebuild and compare every cell
  const b2 = new Board(size);
  for (const [x, y, color] of stones) b2.set(x, y, color);
  let match = true;
  for (let y = 0; y < size; y++)
    for (let x = 0; x < size; x++)
      if (g.board.get(x, y) !== b2.get(x, y)) match = false;
  assert(match, 'round-trip preserves all cells');
}
