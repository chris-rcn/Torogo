function raveScore(child, parentVisits, parentRaveWins, parentRaveVisits, move) {
  if (child.visits === 0) return Infinity;
  const realWR   = child.wins / child.visits;

function raveScoreV1(moveIdx, node) {
  const realV = node.visits[moveIdx];
  const move  = node.legalMoves[moveIdx];
  const raveV = move >= 0 ? node.raveVisits[move] : 0;
  const raveWR = raveV > 0 ? node.raveWins[move] / raveV : 0.5;
  if (realV === 0) {
    return 10 + raveWR;
  }
  const realWR = node.wins[moveIdx] / realV;
  const beta = Math.sqrt(RAVE_EQUIV / (3 * realV + RAVE_EQUIV));
  return (1 - beta) * realWR + beta * raveWR
       + EXPLORATION_C * Math.sqrt(Math.log(node.totalVisits + 1) / realV);
}

function raveScoreV2(moveIdx, node) {
  // raveWins initialized to 0.01;
  // raveVisits initialized to 0.02;
  const realV = node.visits[moveIdx];
  const move  = node.legalMoves[moveIdx];
  const raveW = move === PASS ? 0 : node.raveWins[move];
  const raveV = move === PASS ? 1 : node.raveVisits[move];
  const raveWR = raveW / raveV;
  if (realV === 0) {
    return 10 + raveWR + 0.01 * Math.random();
  }
  const realWR = node.wins[moveIdx] / realV;
  const beta = Math.sqrt(RAVE_EQUIV / (3 * realV + RAVE_EQUIV));
  return (1 - beta) * realWR + beta * raveWR
       + EXPLORATION_C * Math.sqrt(Math.log(node.totalVisits + 1) / realV);
}

function raveScoreV3(moveIdx, node) {
  // wins initialized to 0.01;
  // visits initialized to 0.02;
  // raveWins initialized to 0.01;
  // raveVisits initialized to 0.02;
  const realV = node.visits[moveIdx];
  const move  = node.legalMoves[moveIdx];
  const raveW = move === PASS ? 0 : node.raveWins[move];
  const raveV = move === PASS ? 1 : node.raveVisits[move];
  const raveWR = raveW / raveV;
  const realWR = node.wins[moveIdx] / realV;
  const beta = Math.sqrt(RAVE_EQUIV / (3 * realV + RAVE_EQUIV));
  return 0.001 * Math.random() +
    (1 - beta) * realWR +

}
